# Colors

## Misc

<>
  {Object.keys(theme.color).map(
    (name) =>
      theme.color[name] instanceof ColorValue && (
        <div key={name}>
          <h3>{name}</h3>
          <ColorPreview color={theme.color[name]} />
        </div>
      )
  )}
</>

<>
  {Object.keys(theme.color.variant).map((variantName) => (
    <div key={variantName}>

        <div key={name}>
          <h4>{name}</h4>
          <ColorPreview color={theme.color.variant[variantName][name]} />
        </div>
      ))}
    </div>
  ))}
</>
